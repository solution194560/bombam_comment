    """

          "aborted": 조기 중단 여부 (bool)
        }
    """
    rows = []
    cutoff_date = ...  # 기존 로직
    failed_count = 0
    stats = {"total": 0, "succeeded": 0, "failed": 0, "aborted": False}
    
    for idx, work in enumerate(works):
        stats["total"] += 1
        try:

### 구현

**위치:** `ridi_collector.py` → `_open_book_and_scroll()` 또는 `_collect_reviews()` 함수 (page.goto 호출 부분)

**변경:**

async def _open_book_and_scroll(page, url):
    """
    작품 페이지 접속, 재시도 로직 포함.
    첫 시도: 30초, 실패 시 10초 대기 후 재시도: 30초
    """
    max_attempts = 2
    for attempt in range(1, max_attempts + 1):
        try:
            timeout = 30000 if attempt == 1 else 30000  # 양쪽 다 30초 (총 60초 범위 내)
            await page.goto(url, wait_until="domcontentloaded", timeout=timeout)
            if attempt > 1:
                logger.info(f"  → 재시도로 성공")
            return
        except Exception as e:
            logger.warning(f"  [시도 {attempt}/{max_attempts}] Page.goto 실패: {e}")
            if attempt < max_attempts:
                await asyncio.sleep(10)  # 재시도 전 10초 대기
            else:
                raise  # 마지막 시도도 실패하면 예외 발생

## 수정 4: getUpdates 실패 시 지수 백오프 (bot.py)

### 목표
- 네트워크 장애 중 불필요한 재접속 반복으로 인한 로그 도배 줄임
- 실패가 이어지면 재시도 간격을 점점 늘림

### 구현

**위치:** `bot.py:200~240` (현재 `main()` 함수 또는 폴링 루프)

**현재:**
```python
def main():
    while True:
        try:
            updates = _api("getUpdates", offset=offset, timeout=25)
            if updates is None:
                time.sleep(1)
                continue
        except Exception as e:
            logger.error(f"폴링 오류: {e}")
            time.sleep(1)
            continue
```

**변경:**
```python
def main():
    failure_count = 0
    max_backoff = 60
    log_throttle = 0
    
    while True:
        try:
            updates = _api("getUpdates", offset=offset, timeout=25)
            
            if updates is None:
                failure_count += 1
                # 지수 백오프: 1초 → 2초 → 4초 → 8초 → ... → 60초
                backoff = min(2 ** (failure_count - 1), max_backoff)
                
                # 로그 도배 방지: 매 10회마다만 출력
                log_throttle += 1
                if log_throttle % 10 == 0 or failure_count == 1:
                    logger.warning(f"[getUpdates 실패 {failure_count}회] "
                                   f"{backoff}초 대기 후 재시도...")
                
                time.sleep(backoff)
                continue
            else:
                # 성공 시 카운터 초기화
                if failure_count > 0:
                    logger.info(f"getUpdates 정상 복구 (실패 {failure_count}회 후)")
                failure_count = 0
                log_throttle = 0
                
        except Exception as e:
            failure_count += 1
            backoff = min(2 ** (failure_count - 1), max_backoff)
            logger.error(f"폴링 예외 (실패 {failure_count}회): {e} → {backoff}초 대기")
            time.sleep(backoff)
            continue
```

## 수정 5 (선택): 사전 연결 점검 (ridi_collector.py)

### 목표
- Chromium을 띄우기 전에 리디북스 서버 도달 가능 여부 확인
- 네트워크 장애 중 Chromium 메모리(2GB NAS에서 치명적) 점유 방지

### 구현

**위치:** `ridi_collector.py:698` (gather_recent 함수 시작 부근)

```python
def gather_recent(days=7):
    """최근 N일 댓글 수집 (선택사항: 사전 연결 점검)"""
    
    # [선택] 사전 연결 점검
    if not _check_network_connectivity():
        logger.error("리디북스 서버 도달 불가 → 브라우저 시작 안 함")
        return [], "", {
            "total": 0, "succeeded": 0, "failed": 1, "aborted": True
        }
    
    # ... 기존 로직 (Chromium 시작)

def _check_network_connectivity(retries=2, retry_interval=20):
    """
    리디북스 및 텔레그램 서버 도달 가능 여부 확인 (사전 진단).
    
    Args:
        retries: 재시도 횟수
        retry_interval: 재시도 간격 (초)
    
    Returns:
        True if 접속 가능, False if 모두 실패
    """
    import urllib.request
    
    targets = [
        ("https://ridibooks.com", 10),
        ("https://api.telegram.org", 10),
    ]
    
    for url, timeout in targets:
        success = False
        for attempt in range(1, retries + 1):
            try:
                req = urllib.request.Request(url, method='HEAD')
                with urllib.request.urlopen(req, timeout=timeout) as resp:
                    if resp.code in (200, 403, 404):  # 403/404도 서버 응답이므로 OK
                        success = True
                        break
            except urllib.error.URLError as e:
                if "timed out" in str(e).lower():
                    logger.warning(f"  {url} 시도 {attempt}/{retries} 타임아웃 (대기 {retry_interval}초)")
                    if attempt < retries:
                        time.sleep(retry_interval)
                else:
                    # 타임아웃이 아닌 다른 오류(403 등)는 서버가 응답한 것 → 통과
                    success = True
                    break
        
        if not success:
            logger.error(f"{url} 모두 실패 → 네트워크 장애로 판단")
            return False
    
    logger.info("사전 연결 점검 통과")
    return True
```
